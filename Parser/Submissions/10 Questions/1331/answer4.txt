import java.util.Scanner; //above main method

Scanner in = new Scanner(System.in);
System.out.println("Enter an integer number greater than 0: "
int num = in.nextInt();
int counter = 0;
for(int i = 0; i<=num;i++) {
	if(i%2==1)
		counter++;
}
System.out.println("There are " + counter + " odd integers between 0 and " + num);
in.close();
