import Java.util.Scanner;

Scanner input = new Scanner(System.in);
System.out.println("Enter a number: ");
int num = input.nextInt();
int temp = 0;
int count = 1;
while(count != 0) {
	if(num % (10 * count) == num)
		count = 0;
	temp += num % (10 * count);
	count += 1;
}
System.out.println("Is " + num + " a palindrome?");
if(temp == num)
	System.out.println("true");
else
	System.out.println("false");
