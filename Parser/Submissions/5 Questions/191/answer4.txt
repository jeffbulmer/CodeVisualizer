import java.util.Scanner;
Scanner in = new Scanner (System.in);
System.out.println("Enter an integer number greater than 0: ");
int num = in.nextInt();
int count = 0;
for (int i = 0; i<=num; i++){
	if (i%2!=0){
		count++;
	}
}
System.out.println("There are " + count + " odd integers between 0 and " + num);
