import java.util.*;

public class Main {
	public static void main(String[] args) {
		System.out.println("Enter an integer number greater than 0: ");
		Scanner in = new Scanner(System.in);
		int a = in.nextInt();
		int count = 0;
		if (a%2 !=0) {
			count++;
		}
		else {
			while (a > 0) {
				a = a-2; 
				count++;
			}
		}
		System.out.println("There are " + count + " odd integers between 0 and " + a);
	}
}
