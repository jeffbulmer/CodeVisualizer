import java.util.Scanner;
Scanner in=new Scanner(System.in);
System.out.println(""Enter a word:"");
String a=in.nextLine();
int i=0,j=a.length();
boolean p=false;
while((a.charAt(i)).equalsto(a.charAt(j))){
	if((i=j-1)&&(a.length()%2==0)){
		p=true;
	}else if((i=j-2)&&(a.length()%2==1)){
		p=true;
	}else{
		j--;i++;
	}
}
