import java.util.Scanner;
Scanner in = new Scanner(System.in);
System.out.println(""Enter triangle length: "");
int n = in.nextInt();
for (int i = 1; i <=n ; i++) {
	for (int j = n-i; j >=1; j--) {
    	System.out.print("" "");
   	}
   for (int k = 1; k <= i; k++) {
    	System.out.print(""*""); 
   }
   	System.out.println("""");
