import java.util.Scanner;
Scanner in = new Scanner (System.in);
System.out.println(""Enter a number: "");
int num = in.nextInt();
int temp = num;
int r = 0;
int i = 0;
while (num>0){
i = num%10;
r = (r*10) + i;
num = num/10;}
System.out.println(""Is "" + temp + "" a palindrome?"");
if (temp == r){
System.out.println(""true"");
}else{
System.out.println(""false"");}
