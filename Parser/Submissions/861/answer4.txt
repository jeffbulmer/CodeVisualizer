int input = 0;
int count = 0;

Scanner sc = new Scanner(System.in);
System.out.println(""Enter an integer number greater than 0: "");
input = sc.nextInt();

for(int i = 0; i <= input; i++){
	if(i%2 == 1){
		count++;
	}
}

System.out.println(""There are "" + count + "" odd integers between 0 and "" + input);
sc.close();
